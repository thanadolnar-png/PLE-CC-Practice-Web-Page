if (saved) { try { AppState.checklistProgress = JSON.parse(saved); } catch (e) { AppState.checklistProgress = {}; } } } function saveChecklistProgress() { localStorage.setItem('ospe_checklist_progress', JSON.stringify(AppState.checklistProgress));
